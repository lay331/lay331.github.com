#include<iostream>
using namespace std;
bool g_InvalidInput=false;

int FindTheID(int numbers[],int length)//该算法只能找到出现次数超过 一般的ID，若刚好只出现一半，如0,1,2,1则无能为力
{
	g_InvalidInput=false;
	if(numbers==0||length<=0)
	{
		g_InvalidInput=true;
		return 0;
	}

	int idNumber=0;
	int rptTime=0;
	for(int i=0;i<length;++i)
	{
		if(rptTime==0)
		{
			idNumber=numbers[i];
			rptTime=1;
		}
		else
		{
			if(idNumber==numbers[i])
			{
				rptTime++;
			}
			else
			{
				rptTime--;			
			}

		}
	}
	return idNumber;
}

void Test(int numbers[],int length,int expValue,bool invalidFlag)
{
	int value=FindTheID(numbers,length);
	if(value==expValue&&g_InvalidInput==invalidFlag)
	{
		cout<<"Passed!!"<<endl;
	}
	else
	{
		cout<<"Failed!!"<<endl;
	}
}

int main()
{
int num1[]={2,1,2,1,2};
Test(num1,5,2,false);

int num2[]={2,2,2,1,1};
Test(num2,5,2,false);

int num3[]={1,2,2,2,1};
Test(num3,5,2,false);

int num4[]={1,1,2,2,2};
Test(num4,5,2,false);

Test(NULL,0,0,true);
return 0;
}
